c = sp.run("netstat -ano | findstr " + str(self.port), shell=True, stdout=sp.PIPE, stderr=sp.STDOUT)
res = c.stdout.strip()
if res != b"":
    pid = int(res.split(b' ')[-1])
    try:
        os.kill(pid, signal.SIGTERM)  # Not certain about second parameter
    except Exception as e:
        print(e)
